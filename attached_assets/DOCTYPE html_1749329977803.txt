  <script>
    function toggleSidebar() {
      document.getElementById('sidebar').classList.toggle('open');
      document.getElementById('overlay').classList.toggle('active');
    }
    function showSection(id) {
      document.querySelectorAll('.content-section').forEach(el =>
        el.classList.remove('active')
      );
      document.getElementById(id).classList.add('active');
      toggleSidebar(); // optionally close sidebar after click
    }
    function launchProxy(url) {
      window.open(url, '_blank');
    }
    function toggleBlankMode() {
      const t = document.getElementById('blankToggle');
      t.classList.toggle('active');
      if (t.classList.contains('active')) {
        const w = window.open();
        w.document.write(
          `<base target="_blank"><script>location.replace("${location.href}")<\/script>`
        );
      }
    }
    function toggleAntiClose() {
      const t = document.getElementById('antiCloseToggle');
      t.classList.toggle('active');
      if (t.classList.contains('active')) {
        window.onbeforeunload = () => 'Are you sure you want to leave?';
      } else {
        window.onbeforeunload = null;
      }
    }
    function changeBgColor(color) {
      document.documentElement.style.setProperty('--bg-color', color);
    }
    // close sidebar if clicked on overlay
    document.getElementById('overlay').onclick = toggleSidebar;
  </script>
